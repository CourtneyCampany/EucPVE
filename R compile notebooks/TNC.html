#Read in spot measurements and merge plot design
TNC <- read.csv("TNC.csv")

TNC$Date <- as.Date(TNC$Date, format = "%m/%d/%Y")
TNC$volume <- as.factor(TNC$volume)
#TNC$campaign < as.factor(TNC$campaign)

#data are already plot means for each volume treatmet with SE's
#there are six dates to coincide with gas exchange campaigns


#PLOTTING

#plot bits
gradient <- colorRampPalette(c("red", "blue"))
palette(gradient(7))
pchs <- c(rep(16,6),17)


#tnc1<- subset(TNC, volume == "1000")

#Sugars
windows(8,6)
par(oma = c(0, 1, 0, 0), mgp = c(2.5, 1, 0), cex.axis=1.0, cex.lab=1.3)

with(TNC, plot(Date, sugar_mgperg, cex=1.6, pch=pchs[volume], col=as.factor(volume),
               ylab = expression(Soluble~Sugars~~(mg/g~DWT)),
               ylim=c(0,254) ,xlab=""))
for(i in 1:7){

  dat <- subset(TNC, volume==levels(volume)[i])

  with(dat, arrows(Date, sugar_mgperg, Date, sugar_mgperg+sugars_se, angle=90, length=0.05,col=palette()[i], lwd=2))
  with(dat, arrows(Date, sugar_mgperg, Date, sugar_mgperg-sugars_se, angle=90, length=0.05,col=palette()[i], lwd=2))

  with(dat,points(Date, sugar_mgperg, type='l', lwd=2, col=palette()[i],  pch=pchs[i]))


}
leglab <- levels(TNC$volume)
leglab[7] <- "free"
legend("topright", leglab, pch=pchs,text.font=3, inset=0.0, 
       title=expression(Pot~volume~(l)), col=palette(), bty='n')

box()



dev.copy2pdf(file= "sugars.pdf")
